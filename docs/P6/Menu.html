def cal_arr
  v = []
  val = @alimentos.collect {|x| x.val_en}
  hc = @alimentos.collect {|x| x.gei}
  gr = @gramos.collect {|x| x*0.01}
  
  val.zip(gr).map{|x, y| x * y}
  hc.zip(gr).map{|x, y| x * y}

  val_tot = 0
  val.each { |x| val_tot+=x}
  
  hc_tot = 0
  hc.each { |x| hc_tot+=x}
  v << val_tot
  v << hc_tot
  return v

end

  end
  
  @gei=aux
