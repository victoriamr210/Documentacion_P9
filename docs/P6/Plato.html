def por_glu

  gr = @gramos.collect {|x| x*0.01}
  glu = @alimentos.collect {|x| x.kcal_glu}
  tot = @alimentos.collect {|x| x.val_en}

  kcal_total = 0
  kcal_glu = 0

  for i in (0...tot.size) do
    kcal_glu += (glu[i] *gr[i])
    kcal_total += (tot[i] *gr[i])
  end 

  por = (kcal_glu * 100.00) / kcal_total

end

def to_s

  a ="Nombre:#{@nombre} "
  a << "%proteinas:"
  a << "#{por_pro.ceil(2)} "  
  a << "%lipidos:"  
  a << "#{por_lip.ceil(2)} "  
  a << "%glucidos:"  
  a << "#{por_glu.ceil(2)} "  
  a << "VCT:"  
  a << "#{vct.ceil(2)}"  

end
